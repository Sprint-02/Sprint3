//Localidade dos sensores na aplicação do Arduíno (Porta) e entrada de dados//
#define DHTPIN A0
#define LM35PIN A5
 

DHT dht(DHTPIN, DHT11);

void setup()
// Este código vai ser a primeira função que o arduíno irá executar assim que for ligado (é executada apenas uma vez)
{
  pinMode(DHTPIN, INPUT);
  Serial.begin(9600);
  dht.begin();
}

void loop()
 // Este código vai ser responsável pelo loop dos sensortes //
//O código "float" é responsável na captura do dado gerado pelo sensor//
{
  float dht11_umidade = dht.readHumidity();
  float dht11_temperatura = dht.readTemperature();
  Serial.print(dht11_umidade);
  Serial.print(";");
  Serial.print(dht11_temperatura);
  Serial.print(";");

  
  float lm35_temperatura = analogRead(LM35PIN);
  lm35_temperatura = lm35_temperatura * 0.00488;
  lm35_temperatura = lm35_temperatura * 100;
  Serial.print(lm35_temperatura);
  Serial.print(";");

  
  Serial.println();
  delay(36000000); //Temporizador da captura dos dados//
}
